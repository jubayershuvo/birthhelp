server {
    listen 80;
    server_name udcsheva.com ;

    # Redirect all HTTP to HTTPS (optional but recommended)
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name udcsheva.com;

    ssl_certificate /etc/letsencrypt/live/udcsheva.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/udcsheva.com/privkey.pem; # managed by Certbot

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # Proxy requests to Next.js
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
